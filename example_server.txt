#_Path environment variables is the prefix <DOCUMENT_ROOT>/app/controller
<VirtualHost *:80>
	ServerAdmin webmaster@localhost

	DocumentRoot /var/www
	RewriteEngine on

	RewriteRule ^/(home)$ %{ENV:CONTROLLER_PATH}?controller=home&method=index [L]
	#index paths
	RewriteRule ^/(members)$ %{ENV:INDEX_PATH}&controller=$1 [L]

	#show paths
	RewriteRule ^/(members|comments)/([0-9]+)$ %{ENV:SHOW_PATH}&controller=$1&id=$2 [L]
	
	#new paths
	RewriteRule ^/(members)/new$ %{ENV:NEW_PATH}&controller=$1 [L] =GET

	#create paths
	RewriteRule ^/(members|comments)/create$ %{ENV:CREATE_PATH}&controller=$1 [L] =POST

	RewriteRule .* /index.php
</VirtualHost>
